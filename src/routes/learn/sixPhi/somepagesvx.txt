  onMount(() => {
    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(70, container.clientWidth / container.clientHeight, 0.1, 1000);
    const renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setSize(container.clientWidth, container.clientHeight);
    container.appendChild(renderer.domElement);

    const controls = new OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;
    controls.dampingFactor = 0.05;
    controls.update();

    const axesHelper = new THREE.AxesHelper(2);
    scene.add(axesHelper);

    // Visualize basis vectors
    const origin = new THREE.Vector3(0, 0, 0);
    for (const basis of SixPhiVector.basis) {
      const dir = new THREE.Vector3(...basis.toFloatArray()).normalize();
      const arrow = new THREE.ArrowHelper(dir, origin, 1.5, 0x888888);
      scene.add(arrow);
    }

    // Visualize computed vector
    debugger;
    const target = new THREE.Vector3(...vector.toCartesian());
    const vecArrow = new THREE.ArrowHelper(target.clone().normalize(), origin, target.length(), 0xff0000);
    scene.add(vecArrow);

    camera.position.set(3, 3, 5);
    camera.lookAt(0, 0, 0);

    const animate = () => {
      requestAnimationFrame(animate);
      controls.update();
      renderer.render(scene, camera);
    };
    animate();

    return () => {
      container.removeChild(renderer.domElement);
    };
  });
</script>
